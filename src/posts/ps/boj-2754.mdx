score = input()

grade = score[0]
sub_grade = score[1] if len(score)>=2 else ""

score = ["F", "D", "C", "B", "A"].index(grade)
if sub_grade == "+":
    score += 0.3
if sub_grade == "-":
    score -= 0.3

print("{:.1f}".format(score))
