```python
n = int(input())
inputs = [input() for _ in range(n)]

for i in inputs:
    cases = i.split(" ")
    num = float(cases[0])
    for case in cases[1:]:
        if case=="@":
            num *= 3
        if case=="%":
            num += 5
        if case=="#":
            num -= 7
    print("{:.2f}".format(num))
```
