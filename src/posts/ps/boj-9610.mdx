```python
n = int(input())
points = [input() for _ in range(n)]

axis = 0
q = [0, 0, 0, 0]
for point in points:
    a, b = map(int, point.split(" "))
    if a==0 or b==0:
        axis += 1
    if a>0 and b>0:
        q[0] += 1
    if a<0 and b>0:
        q[1] += 1
    if a>0 and b<0:
        q[3] += 1
    if a<0 and b<0:
        q[2] += 1

for idx, n in enumerate(q):
    print("Q{}: {}".format(idx+1, n))
print("AXIS: {}".format(axis))

```
