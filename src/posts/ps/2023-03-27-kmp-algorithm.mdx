### kmp

def fail():
    f = [0 for _ in range(len(s))]
    j = 0
    for i in range(1, len(s)):
        while j > 0 and s[i] != s[j]:
            j = f[j-1]
        if s[i] == s[j]:
            j += 1
            f[i] = j
    return f


def kmp():
    _fail = fail()
    i, j = 0, 0

        if j == len(s):
